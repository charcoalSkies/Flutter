### Column MainAxisAlignment

``` dart 
import 'package:flutter/material.dart';

class HomeScreen extends StatelessWidget {
	const HomeScreen({super.key});

	@override 
	Widget build(BuildContext context){
		return Scaffold(
			body: Container(
				color: Colors.black,
				child: Column(
					// MainAxisAlignment : 추축 정렬
					// start : 시작 부분에 정렬 (기본값)
					// end : 끝 부분에 정렬
					// center : 가운데 정렬
					// spaceBetween : 위젯과 위젯 사이가 동일하게 배치
					// spaceEvenly : 위젯을 같은 간격으로 배치하지만 끝과 끝도 
														//위젯이 아닌 빈 간격으로 시작
					// spaceAround : spaceEvenly + 끝과 끝의 간격은 1 / 2
					mainAxisAlignment: MainAxisAlignment.spaceAround
					children: [
						Container(
							color: Colors.red,
							width: 50.0,
							height: 50.0,
						),
						Container(
							color: Colors.orange,
							width: 50.0,
							height: 50.0,
						),
						Container(
							color: Colors.yello,
							width: 50.0,
							height: 50.0,
						),
						Container(
							color: Colors.green,
							width: 50.0,
							height: 50.0,
						),
					],
				)	
			),
		);
	}
}
```

### Column CrossAxisAlignment

``` dart
import 'package:flutter/material.dart';

class HomeScreen extends StatelessWidget {
	const HomeScreen({super.key});

	@override 
	Widget build(BuildContext context){
		return Scaffold(
			body: Container(
				color: Colors.black,
				// 사용중인 휴대폰의 사이즈 가져오기(넗이 또는 길이)
				// 강제로 Column이 휴대폰 넓이 만큼 차지하도록 조절
				width: MediaQuery.of(context).size.width,
				child: Column( // Column이 차지할 수 있는 최대 크기
					mainAxisAlignment: MainAxisAlignment.spaceAround

					// CrossAxisAlignment - 반대축 정렬
					// start : 시작
					// end : 끝
					// center : 가운데 (기본값)
					// stretch : 최대한으로 늘린다
					crossAxisAlignment: CrossAxisAlignment.strech,
					children: [
						Container(
							color: Colors.red,
							width: 50.0,
							height: 50.0,
						),
						Container(
							color: Colors.orange,
							width: 50.0,
							height: 50.0,
```

### Expanded & Flexible

``` dart 
class HomeScreen extends StatelessWidget{
	const HomeScreen({super.key}):

	@override
	Widget build(BuildContext context){
		return Scaffold(
			body: Container(
				color: Colors.black 
				child: Column(
					mainAxisAlignment: MainAxisAlignment.start,
					mainAxisSize: mainAxisSize.max,
					children:[
					
						// Expanded / Flexible (Row 나 Column 의 children에서만 
						//										사용할 수 있다!!!)



						// Expanded : 남아있는 공간을 최대한으로 Expanded 끼리 나누어 차지
						// flex : Expanded의 파라미터로 
						//          나머지 공간을 차지하는 비율(flex로 Expanded 비율 조절)

						// Flexible : 일단 컨테이너의 높이와 넓이 만큼 공간을 차지
						// 만약 child안에 있는 위젯이 그 공간을 전부 차지하지 않을 경우 
						//                                           남는공간을 버림
						// flex : Flexible의 파라미터로 
						//          나머지 공간을 차지하는 비율(flex로 Expanded 비율 조절)

						Flexible(
							flex:2,
							child: Container(
								color: Colors.red
								width: 50.0,
								height: 50.0,
							),
						),
						Flexible(
							flex:2,
							child: Container(
								color: Colors.red
								width: 50.0,
								height: 50.0,
							),
						),
						Expanded(
							flex:2,
							child: Container(
								color: Colors.red
								width: 50.0,
								height: 50.0,
							),
						),
						Expanded(
							flex:2,
							child: Container(
								color: Colors.red
								width: 50.0,
								height: 50.0,
							),
						),
					],
				),	
			),
		);
	}
}
```
